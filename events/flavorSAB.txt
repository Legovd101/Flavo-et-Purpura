# Sabaria Offers Their Hand in Marriage
country_event = {
	id = flavor_SAB.3
	title = "flavor_SAB.3.t"
	desc = "flavor_SAB.3.d"
	picture = FEAST_eventPicture
	
	is_triggered_only = yes

	
	immediate = {
		add_prestige = 5
	}
	
	option = { 
		name = "flavor_SAB.3.a" # Let us bind their dynasty to ours
		if = {
			limit = { tag = GAL }
			break_union = overlord
			FROM = { create_union = ROOT }
			override_country_name = CALLIA
			LUS = {
				country_event = {
					id = flavor_SAB.4
				}
			}
		}
		if = {
			limit = { tag = NAV }
			FROM = { create_union = ROOT }
		}
		else_if = {
			limit = {
				has_country_flag = already_recieved_offer
			}
			FROM = {
				create_union = ROOT
			}
		}
		else_if = {
			limit = { tag = LUS }
			set_country_flag = already_recieved_offer
			force_remove_rival = FROM
			create_marriage = FROM
			add_opinion = {
				who = LUS
				modifier = grateful_for_aid
			}
			define_heir = {
				dynasty = SAB
			}
			define_consort = {
				dynasty = SAB
			}
		}
	}
	
	option = { 
		name = "flavor_SAB.3.b" # Are they crazy!?
		FROM = {
			country_event = {
				id = flavor_SAB.5
			}
		}
		ai_chance = {
			factor = 0
		}
	}
}
