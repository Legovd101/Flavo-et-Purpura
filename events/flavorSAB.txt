# Absorption of the Asturean Throne

	id = flavor_sab.1
	title = "flavor_sab.1.t"
	desc = "flavor_sab.1.d"
	picture = GOOD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SAB
		ASU = { exists = no }
	}
	mean_time_to_happen = {
		days = 0
	}

	
	immediate = {
		add_prestige = 5
	}
	
	option = { 
		name = "flavor_sab.1.a" # Integrate the Astureans into our court
		add_country_modifier = {
			name = ara_maritime_administration_modifier

		}

		basque_country = {
			add_permanent_claim = ROOT
		}
		castille_area = {
			add_permanent_claim = ROOT
		}
	}
	
	option = { 
		name = "flavor_sab.1.b" # The Astureans are merely confused Sabarians, is all!
		add_country_modifier = {
			name = arm_national_awakening
			duration = 18250
		}

			add_base_manpower = 1
			add_base_production = 1
			add_base_tax = 1
		}
		basque_country = {
			add_permanent_claim = ROOT
		}
		castille_area = {
			add_permanent_claim = ROOT
		}
	}
}

# The Iberian Wedding
country_event = {
	id = flavor_sab.2
	title = "flavor_sab.2.t"
	desc = "flavor_sab.2.d"
	picture = GOOD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SAB
		GAL = {
			has_heir = no
			has_opinion = {
				who = ROOT
				value = 50
			}
		}
	}
	mean_time_to_happen = {
		years = 1
		modifier = {
			factor = 0.5
			reverse_has_opinion = {
				who = GAL
				value = 100
			}
		}
	}

	
	immediate = {
	}
	
	option = { 
		name = "flavor_sab.2.a" # Let us bind their dynasty to ours.
		add_prestige = -10
		GAL = {
			country_event = {
				id = flavor_sab.3
				days = 10
			}
		}
	}
	
	option = { 
		name = "flavor_sab.2.b" # We musn't sully our ties to Lusitania, we shall marry them instead.
		LUS = {
			country_event = {
				id = flavor_sab.3
				days = 10
			}
		}
	}
	
	option = { 
		name = "flavor_sab.2.c" # To hell with them both! Their land is forfeit!
		declare_war_with_cb = {
			casus_belli = cb_annex
			who = LUS
		}
		trigger = {
			is_rival = LUS
		}
	}
	
	option = { 
		name = "flavor_sab.2.e" # No.
		galicia_area = {
			add_permanent_claim = ROOT
		}
		beieras_area = {
			add_permanent_claim = ROOT
		}
	}
}

# Sabaria Offers Their Hand in Marriage
country_event = {
	id = flavor_sab.3
	title = "flavor_sab.3.t"
	desc = "flavor_sab.3.d"
	picture = FEAST_eventPicture
	
	is_triggered_only = yes

	
	immediate = {
		add_prestige = 5
	}
	
	option = { 
		name = "flavor_sab.3.a" # Let us bind their dynasty to ours
		if = {
			limit = {
				tag = GAL
			}
			break_union = overlord
			FROM = {
				create_union = ROOT
			}
			LUS = {
				country_event = {
					id = flavor_sab.4
				}
			}
		}
		if = {
			limit = {
				tag = LUS
			}
			create_marriage = FROM
			add_opinion = {
				who = LUS
				modifier = grateful_for_aid
			}
			define_heir = {
				dynasty = SAB
			}
			define_consort = {
				dynasty = SAB
			}
		}
	}
	
	option = { 
		name = "flavor_sab.3.b" # Are they crazy!?
		FROM = {
			country_event = {
				id = flavor_sab.5
			}
		}
	}
}
