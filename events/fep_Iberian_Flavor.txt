# At the doors of Rome
country_event = {
	id = fep_iberian_flavor.3
	title = fep_iberian_flavor.3.t
	desc = fep_iberian_flavor.3.d
	picture = ICONS_IN_CHURCH_eventPicture
	
	hidden = yes
	
	is_triggered_only = yes

	option = {
		name = fep_iberian_flavor.3.a
		every_province = {
			limit = {
				region = italy_region
				NOT = {
					AND = {
						area = corsica_sardinia_area
						area = corsica_area
						area = sicily_area
						area = western_sicily_area
					}
				}
			}
			if = {
				limit = {
					OR = {
						area = calabria_area
						area = apulia_area
						area = naples_area
						area = lazio_area
						area = central_italy_area
						area = tuscany_area
					}
				}
				add_core = PAP
			}
			else = {
				add_core = TUS
			}
		}
		release = PAP
		release = TUS
		create_subject = {
			subject_type = imperial_province
			who = PAP
		}
		PAP = {
			change_country_color = {
				country = ROOT
			}
			override_country_name = ITALIA_PONTIFICALIA
		}
		TUS = {
			change_country_color = {
				country = ROOT
			}
			override_country_name = ITALIA_ANNONARIA
		}
		create_subject = {
			subject_type = imperial_province
			who = TUS
		}
	}

	option = {
		name = fep_iberian_flavor.3.b
		add_prestige = 20
		add_country_modifier = {
			name = hab_conquest_of_italy
			duration = -1
		}
		118 = {
			add_great_project_tier = {
				type = stpeters_cathedral
				tier = 1
			}
		}
	}
}
