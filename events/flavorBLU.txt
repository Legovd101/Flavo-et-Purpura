# Death of Kaidu Khan
country_event = {
	id = flavor_blu.1
	title = "flavor_blu.1.t"
	desc = "flavor_blu.1.d"
	picture = DEATH_OF_HEIR_eventPicture
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		tag = BLU
		NOT = { has_global_flag = death_of_kaidu_khan }
	}

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_global_flag = death_of_kaidu_khan
		every_subject_country = {
			add_liberty_desire = 20
		}
		add_government_power = { # sets the power to the specified amount
			mechanic_type = mongol_yoke
			power_type = stranglehold
			value = -25
		}
	}
	
	option = { 
		name = "flavor_blu.1.a" # May the Pax Mongolica ever thrive
		custom_tooltip = flavor_blu.1.a.tt
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { OR = { tag = ROOT tag = GRE } }
					OR = {
						is_subject_of = ROOT
						is_neighbor_of = ROOT
						any_neighbor_country = {
							is_subject_of = ROOT
						}
					}
				}
				country_event = { id = flavor_blu.2 }
			}
			every_country = {
				limit = {
					OR = {
						is_subject_of_type_with_overlord = {
							who = FROM
							type = tributary_state
						}
					}
				}
				country_event = { 
					id = flavor_blu.4 
					days = 1825
					random = 90
				}
			}
		}
	}
}

# A chance at freedom
country_event = {
	id = flavor_blu.4
	title = "flavor_blu.4.t"
	desc = "flavor_blu.4.d"
	picture = DEATH_OF_HEIR_eventPicture
	
	is_triggered_only = yes

	trigger = {
		is_subject_of_type_with_overlord = {
			who = BLU
			type = tributary_state
		}
		NOT = { war_with = BLU }
	}

	option = {
		name = "flavor_blu.4.a" # This is our chance to finally shake the yoke off our necks!
		trigger = {
			is_subject_of = BLU
		}
		BLU = {
			country_event = { 
				id = flavor_blu.3 
				days = 30
				random = 60
			}
		}
		ai_chance = { 
			factor = 50
			modifier = {
				factor = 50
				NOT = { liberty_desire = 50 }
				liberty_desire = 25
			}
			modifier = {
				factor = 0
				NOT = { liberty_desire = 25 }
			}
		}
	}
	option = {
		name = "flavor_blu.4.b" # I don't feel confident in our ability to take on the Mongols...
		country_event = { 
			id = flavor_blu.4 
			days = 1825
			random = 90
		}
		ai_chance = { 
			factor = 50
			modifier = {
				factor = 0
				overlord = {
					is_at_war = yes
				}
			}
		}
	}
}

# The Dissolution of the Blue Horde

	desc = "flavor_blu.5.d"

		name = "flavor_blu.5.a" # We stray further from Ogedei's legacy every day...

		hidden_effect = {
			every_province = {
				limit = {

						area = macedonia_area

				remove_core = ROOT

			}

				limit = {

						area = bosnia_area
