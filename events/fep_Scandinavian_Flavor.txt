# Heir to the Rus'
country_event = {
	id = fep_scandinavian_flavor.4
	title = fep_scandinavian_flavor.4.t
	desc = fep_scandinavian_flavor.4.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	is_triggered_only = yes
	
	#Keep Russia under direct rule
	option = {
		name = fep_scandinavian_flavor.4.a
		add_adm_power = 50
		add_mil_power = 50
		russia_region = {
			limit = { country_or_non_sovereign_subject_holds = ROOT }
			add_province_modifier = {
				name = eng_english_rule_over_ireland
				duration = 9125
			}
		}
		ruthenia_region = {
			limit = { country_or_non_sovereign_subject_holds = ROOT }
			add_province_modifier = {
				name = eng_english_rule_over_ireland
				duration = 9125
			}
		}
		ai_chance = {
			factor = 1
		}
	}
	
	#Rule over Russia in a Personal Union
	option = {
		name = fep_scandinavian_flavor.4.b
		add_adm_power = 50
		add_dip_power = 50
		custom_tooltip = fep_scandinavian_flavor.4.b.tt
		hidden_effect = {
			russia_region = {
				limit = { country_or_non_sovereign_subject_holds = ROOT }
				if = {
					limit = { NOT = { owned_by = ROOT } }
					cede_province = ROOT
				}
				add_core = RUS
				if = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
			ruthenia_region = {
				limit = { country_or_non_sovereign_subject_holds = ROOT }
				if = {
					limit = { NOT = { owned_by = ROOT } }
					cede_province = ROOT
				}
				add_core = RUS
				if = {
					limit = { is_core = ROOT }
					remove_core = ROOT
				}
			}
			release = RUS
			RUS = {
				override_country_name = kievriket
				change_religion = ROOT
				change_government = monarchy
				# add_government_reform = irish_highkingdom_reform
				add_adm_power = 112
				add_dip_power = 112
				add_mil_power = 112
				add_treasury = 136
				add_manpower = 8
				add_sailors = 3000
				add_army_tradition = 5
			}
			create_subject = {
				subject_type = personal_union
				who = RUS
			}
		}
		add_country_modifier = {
			name = swe_heir_to_rurik
			duration = -1
		}
		ai_chance = {
			factor = 0
		}
	}
}
